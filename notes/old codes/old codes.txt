//percorrendo o mapa desenhando os tiles com as sprites corretas nos tiles definidos
for(var _i = 0; _i < MAP_WIDTH; _i++){
    for(var _j = 0; _j < MAP_HEIGHT; _j++){
        //calcula a posição correta e tamanho de cada tile
        var _pos_tile_x = _i * TILE_SIZE;
        var _pos_tile_y = _j * TILE_SIZE;
        
        //se a posição for parede
        if(map_dungeon[_i][_j] == TILE_WALL){
            //desenha a sprite de parede
            draw_sprite(spr_tile_wall, 0, _pos_tile_x, _pos_tile_y);
        }
        
        //se a posição for chão
        if(map_dungeon[_i][_j] == TILE_FLOOR){
            //desenha a sprite do chão
            draw_sprite(spr_tile_floor, 0, _pos_tile_x, _pos_tile_y);
        }
    }
}
